PREFIX ns: <http://schema.org/>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> 
PREFIX rdfs:<http://www.w3.org/2001/XMLSchema#>
PREFIX owl: <http://www.owl-ontologies.com/unnamed.owl#>


SELECT ?ville ?name ?available_bikes ?available_bikes_stands ?bike_stands
WHERE {
	 ?station rdf:type ns:station .
	 ?station ns:name ?name .
  	?station ns:ville ?ville .	
	 ?station ns:available_bikes ?available_bikes .
	 ?station ns:available_bikes_stands ?available_bikes_stands .
	 ?station ns:bike_stands ?bike_stands .
  FILTER(?ville="Montpellier") .
  
	   	
}
